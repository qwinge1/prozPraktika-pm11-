-- Представление для учеников (только свои оценки)
CREATE   VIEW [dbo].[Представление_Ученик]
AS
SELECT 
    у.[ФИО] AS Ученик,
    п.[Название] AS Предмет,
    о.[Оценка],
    о.[Дата_получения],
    о.[Тип_контроля]
FROM [Оценки] о
JOIN [Учащиеся] у ON о.[ID_ученика] = у.[ID_ученика]
JOIN [Предметы] п ON о.[ID_предмета] = п.[ID_предмета];
GO

/****** Object:  StoredProcedure [dbo].[Генерация_Оценок]    Script Date: 23.05.2026 13:21:15 ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE   PROCEDURE [dbo].[Генерация_Оценок]
AS
BEGIN
    DECLARE @student_id INT, @subject_id INT, @date DATE, @grade INT, @control VARCHAR(50);
    DECLARE student_cursor CURSOR FOR SELECT [ID_ученика] FROM [Учащиеся];
    
    OPEN student_cursor;
    FETCH NEXT FROM student_cursor INTO @student_id;
    
    WHILE @@FETCH_STATUS = 0
    BEGIN
        DECLARE subject_cursor CURSOR FOR SELECT TOP 5 [ID_предмета] FROM [Предметы] ORDER BY NEWID();
        OPEN subject_cursor;
        FETCH NEXT FROM subject_cursor INTO @subject_id;
        
        WHILE @@FETCH_STATUS = 0
        BEGIN
            DECLARE @count INT = 3 + ABS(CHECKSUM(NEWID())) % 4;
            DECLARE @i INT = 0;
            WHILE @i < @count
            BEGIN
                SET @date = DATEADD(day, ABS(CHECKSUM(NEWID())) % 365, '2025-09-01');
                SET @grade = 2 + ABS(CHECKSUM(NEWID())) % 4;
                SET @control = CHOOSE(1 + ABS(CHECKSUM(NEWID())) % 3, 'текущая', 'контрольная', 'самостоятельная');
                
                INSERT INTO [Оценки] ([ID_ученика], [ID_предмета], [Дата_получения], [Оценка], [Тип_контроля])
                VALUES (@student_id, @subject_id, @date, @grade, @control);
                
                SET @i = @i + 1;
            END
            FETCH NEXT FROM subject_cursor INTO @subject_id;
        END
        CLOSE subject_cursor;
        DEALLOCATE subject_cursor;
        
        FETCH NEXT FROM student_cursor INTO @student_id;
    END
    CLOSE student_cursor;
    DEALLOCATE student_cursor;
END;
